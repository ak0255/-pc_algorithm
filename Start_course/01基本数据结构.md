火车进栈

火车编号1-n

```
出站<——    <——进站
       |车|
       |站|
       |__|
```

每次只能进站一个或出站一个

现在请你按《字典序》输出前 20 种可能的出栈方案。

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<ll, ll> Pll;

int n, cnt = 20；
int	state3 = 1;//还没进站的，因为按照1-n，所以直接定义一个整数1表示排在第一个的
vector<int> state1;//已经出站的
stack<int> state2;//还在站内的

void dfs()
{
    if (!cnt)
        return;
    if (state1.size() == n)//如果已经有n个出栈了，那么输出这n个
    {
        for (auto i : state1)
            cout << i;
        cout << endl;
        cnt--;
    }
    if (state2.size())//如果站内还有元素
    {
        state1.push_back(state2.top());//出栈
        state2.pop();//出栈
        dfs();//dfs
        state2.push(state1.back());//回溯
        state1.pop_back();//回溯
    }
    if (state3 <= n)//如果还有没进站的
    {
        state2.push(state3);//进栈
        ++state3;//车编号加1
        dfs();//dfs
        --state3;//回溯
        state2.pop();//回溯
    }
}

int main()
{
    // freopen("in.txt", "r", stdin);
    std::ios::sync_with_stdio(false);
    cin >> n;
    dfs();
    return 0;
}
```

**字符串哈希**

```cpp
typedef unsigned long long ULL;
ULL h[N], p[N]; // h[k]存储字符串前k个字母的哈希值, p[k]存储 P^k mod 2^64

// 初始化
p[0] = 1;
for (int i = 1; i <= n; i ++ )
{
    h[i] = h[i - 1] * P + str[i];
    p[i] = p[i - 1] * P;
}

// 计算子串 str[l ~ r] 的哈希值
ULL get(int l, int r)
{
    return h[r] - h[l - 1] * p[r - l + 1];
}
```

### Trie

**俗称：字典树**

- **统计前缀**

> 给定 *N* 个字符串 *S*1,*S*2…*SN*，接下来进行 *M* 次询问，每次询问给定一个字符串 *T*，求 *S*1∼*SN* 中有多少个字符串是 *T* 的前缀。

```cpp
int n, m, trie[N][26], cnt = 1, End[N];//End数组记录字符串结尾

void insert(string str)//前n个字符串加入树
{
    int dep = 0;
    for (register int i = 0; i < str.length(); ++i)
    {
        if (trie[dep][str[i] - 'a'] == 0)
        {
            trie[dep][str[i] - 'a'] = ++cnt;
        }
        dep = trie[dep][str[i] - 'a'];
    }
    End[dep]++;//该点有多少个字符串结尾
}
int search(string str){//后m个字符串进行询问
    int dep = 0, res = 0;
    for(int i = 0;i < str.size();i++){
        dep = trie[dep][str[i] - 'a'];
        if(dep == 0) return res;
        res += End[dep];
    }
    return res;
}
```

- **最大异或值**

```cpp
int tr[N * 31][2], cnt[N * 31], idx;
void insert(int x, int c){//x为数，c表示建立或移除，建树c == 1，删除c == -1
    int p = 0;
    for(int i = 30;i >= 0;i--){
        int u = x >> i & 1;
        if(!tr[p][u]) tr[p][u] = ++idx;
        p = tr[p][u];
        cnt[p] += c;
    }
}
int query(int x){//查询这个数与前面的数的最大异或值
    int res = 0, p = 0;
    for(int i = 30;i >= 0;i--){
        int u = x >> i & 1;
        if(cnt[tr[p][!u]]) p = tr[p][!u], res = res * 2 + 1;
        //假如我这一位是0，那么我看有没有这一位是1的，有就用，没有就只能用也是0的那一个
        else p = tr[p][u], res = res * 2;
    }
    return res;
}

insert(0, 1);//区间求最大异或时，加入一个0，方便边界前的计算
```

**拓展**

**表达式计算**

> 给出一个表达式,其中运算符仅包含`+,-,*,/,^`（加 减 乘 整除 乘方）要求求出表达式的最终值。
>
> 数据可能会出现括号情况，还有可能出现多余括号情况。
>
> 数据保证不会出现大于或等于2^31的答案。
>
> 数据可能会出现负数情况。

#### 输入样例：

> (2+2)^(1+1)

#### 输出样例：

> 16

括号栈模拟，可使用大部分计算器功能

```cpp
#include <iostream>
#include <cstring>
#include <algorithm>
#include <stack>
using namespace std;

typedef long long ll;
stack<ll> nums;
stack<char> ops;

void cal(){
    char c = ops.top();
    ops.pop();
    ll a = nums.top();
    nums.pop();
    ll b = nums.top();
    nums.pop();
    ll res = 0;
    if(c == '+') res = b + a;
    else if(c == '-') res = b - a;
    else if(c == '*') res = b * a;
    else if(c == '/') res = b / a;
    else{
        res = 1;
        while(a--){
            res *= b;
        }
    }
    nums.push(res);
}

int main(){
    string cc;
    string str;
    string left;
    getline(cin, cc);
    for(int i = 0;i <= cc.size();i++) left += '(';//防止括号不匹配的情况
    cc = left + cc + ')';
    for(int i = 0;i < cc.size();i++)
        if(cc[i] != ' ') str += cc[i];//排除所有空格

    // cout << str << endl;
    for(int i = 0;i < str.size();i++){
        // if(str[i] == ' ') continue;
        if(str[i] <= '9' && str[i] >= '0'){
            ll j = i, t = 0;
            while(str[j] <= '9' && str[j] >= '0'){
                t = t * 10 + str[j] - '0';
                j++;
            }
            i = j - 1;
            nums.push(t);
        }
        else    
        {
            char c = str[i];
            if(c == '+' || c == '-'){
                if(c == '-' && !(str[i - 1] <= '9' && str[i - 1] >= '0' || str[i - 1] == ')')){
                    ll j = i + 1, t = 0;
                    while(str[j] <= '9' && str[j] >= '0'){
                        t = t * 10 + str[j] - '0';
                        j++;
                    }
                    i = j - 1;
                    nums.push(-t);
                    
                    j = i + 1;
                    while(str[j] == ' '){
                        j++;
                    }
                    if(str[j] == '(') ops.push(c);
                    
                }
                else{
                    while(ops.top() != '(') cal();
                    ops.push(c);
                }
            }else if(c == '*' || c == '/'){
                while(ops.top() == '*' || ops.top() == '/' || ops.top() == '^') cal();
                ops.push(c);
            }else if(c == '^'){
                while(ops.top() == '^') cal();
                ops.push(c);
            }else if(c == ')'){
                while(ops.top() != '(') cal();
                ops.pop();
            }else if(c == '('){
                ops.push(c);
            }
        }
    }
    cout << nums.top() << endl;
    return 0;
}
```
